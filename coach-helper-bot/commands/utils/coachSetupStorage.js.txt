import fs from "fs";
import path from "path";

const tempDir = "data/tempSetup";

if (!fs.existsSync(tempDir)) {
  fs.mkdirSync(tempDir, { recursive: true });
}

export function getTempPath(coachId) {
  return path.join(tempDir, `${coachId}.json`);
}

export function loadTempProfile(coachId) {
  const p = getTempPath(coachId);
  if (!fs.existsSync(p)) return {};
  return JSON.parse(fs.readFileSync(p, "utf8"));
}

export function saveTempProfile(coachId, data) {
  const p = getTempPath(coachId);
  fs.writeFileSync(p, JSON.stringify(data, null, 2));
}

export function deleteTempProfile(coachId) {
  const p = getTempPath(coachId);
  if (fs.existsSync(p)) fs.unlinkSync(p);
}
