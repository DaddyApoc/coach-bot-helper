import { SlashCommandBuilder, EmbedBuilder } from "discord.js";
import { getWallet } from "../../utils/wallet.js";
import { loadSessions } from "../../utils/sessions.js";

export default {
  data: new SlashCommandBuilder()
    .setName("student-dashboard")
    .setDescription("View your student dashboard"),

  async execute(interaction) {
    const userId = interaction.user.id;
    const wallet = getWallet(userId);
    const sessions = Object.values(loadSessions()).filter(
      s => s.studentId === userId
    );

    const active = sessions.filter(s => s.status !== "completed");
    const completed = sessions.filter(s => s.status === "completed");

    const embed = new EmbedBuilder()
      .setTitle(`${interaction.user.username}'s Student Dashboard`)
      .setColor("Blue")
      .addFields(
        {
          name: "Wallet Balance",
          value: `$${wallet.balance.toFixed(2)}`,
          inline: true,
        },
        {
          name: "Total Transactions",
          value: `${wallet.history.length}`,
          inline: true,
        },
        {
          name: "Active Sessions",
          value: `${active.length}`,
          inline: true,
        },
        {
          name: "Completed Sessions",
          value: `${completed.length}`,
          inline: true,
        }
      )
      .setTimestamp();

    await interaction.reply({ embeds: [embed], ephemeral: true });
  },
};
